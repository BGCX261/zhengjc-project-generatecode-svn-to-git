*================================常用宏定义=================================*

*&---------------------------------------------------------------------*
*&      DEFINE  CON_OPER
*&---------------------------------------------------------------------*
*       条件运算 宏定义 类似 A = C ? D : E.
*----------------------------------------------------------------------*
*  -->  &1        条件：左数据
*  -->  &2        运算符
*  -->  &3        条件：右数据
*  -->  &4        true  获取的数据
*  -->  &5        false 获取的数据
*  -->  &6        判断后返回的数据
*----------------------------------------------------------------------*
  DEFINE con_oper.
    if &1 &2 &3.
      &6 = &4.
    else.
      &6 = &5.
    endif.
  END-OF-DEFINITION.


*&---------------------------------------------------------------------*
*&      DEFINE  def_range
*&---------------------------------------------------------------------*
*      RANGE
*----------------------------------------------------------------------*
*  -->  &1        RANGE的变量
*  -->  &2        sigm值
*  -->  &3        option值
*  -->  &4        low值
*  -->  &5        high值
*----------------------------------------------------------------------*
* For example：
* RANGES: R_KTOKS FOR SKA1-KTOKS.        " 总帐科目组
* DEF_RANGE R_KTOKS 'I' 'EQ' 'CASH' '' .
*----------------------------------------------------------------------*
  DEFINE def_range.
    clear &1.
    &1-sign = &2.
    &1-option = &3.
    &1-low = &4.
    &1-high = &5.
    append &1.
  END-OF-DEFINITION.


*&---------------------------------------------------------------------*
*&      Select_DATA_TO_WA
*&---------------------------------------------------------------------*
*       结构的动态查询
*----------------------------------------------------------------------*
*  -->  &1        查询字段
*  -->  &2        查询的表
*  -->  &3        填写的内表
*  -->  &4        查询条件
*----------------------------------------------------------------------*
  DEFINE select_data_to_wa.

    select &1
    from &2
    into corresponding fields of &3
    where (&4).
      exit.
    endselect.

  END-OF-DEFINITION.


*&---------------------------------------------------------------------*
*&      Select_DATA_TO_WA
*&---------------------------------------------------------------------*
*       变量的动态查询
*----------------------------------------------------------------------*
*  -->  &1        查询字段
*  -->  &2        查询的表
*  -->  &3        填写的字段
*  -->  &4        查询条件
*----------------------------------------------------------------------*
  DEFINE select_data_to_variant.

    select &1
    from &2
    into &3
    where (&4).
      exit.
    endselect.

  END-OF-DEFINITION.


*&---------------------------------------------------------------------*
*&      Select_DATA_TO_WA
*&---------------------------------------------------------------------*
*       内表的动态查询
*----------------------------------------------------------------------*
*  -->  &1        查询字段
*  -->  &2        查询的表
*  -->  &3        填写的字段
*  -->  &4        查询条件
*----------------------------------------------------------------------*
  DEFINE select_data_to_variant.

    select &1
    from &2
    into corresponding fields of table &3
    where (&4).
    exit.
  endselect.

END-OF-DEFINITION.
